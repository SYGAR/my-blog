    # webpack.config.js中的配置

    // 引入一个包
const path = require('path')

// webpack中所有的配置信息都应该写在module.exports中
module.exports = {
  // 指定入口文件
  entry: './src/index.ts',
  // 指定打包文件所在的目录
  output: {
    // 指定打包后的目录
    // __dirname 是当前文件夹的绝对路径   __filename是当前文件的绝对路径
    path: path.resolve(__dirname, 'dist'),
    // 打包后的文件名
    filename: 'bundle.js', 
    // 不使用箭头函数
    environment: {
      arrowFunction: false
    },
  },
  // 指定webpack打包时要使用的模块
  module: {
    // 指定要加载的规则
    rules: [
      {
        // test指定的是规则生效的文件
        test: /\.ts$/,
        // 使用哪些loader进行处理
        use: 'ts-loader',
        // 要排除的文件
        exclude: /node_modules/,
      },
    ],
  },
  mode: 'development'
}

    # 使用babel

    cnpm i -D babel-loader @babel/core @babel/preset-env babel-loader core-js
    1

    配置babel

     module: {
    // 指定要加载的规则
    rules: [
      {
        // test指定的是规则生效的文件
        test: /\.ts$/,
        // 使用哪些loader进行处理
        use: [{
          // 设置加载器
          loader: 'babel-loader',
          // 设置babel
          options: {
            // 设置预定义的环境
            presets: [
              [
                // 指定环境的插件
                '@babel/preset-env',
                // 配置信息
                {
                  // 要兼容的目标浏览器
                  targets: {
                    "chrome": "88",
                    "ie": "11"
                  },
                  // 指定corejs的版本
                  "corejs": "3",
                  // 使用corejs的方式 "usage" 表示按需加载
                  "useBuiltIns": "usage"
                }
            ]
          ]
          }
        }, 'ts-loader'],
        // 要排除的文件
        exclude: /node_modules/,
      },
    ],
  }
